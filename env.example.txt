# ── Discord Bot ────────────────────────────────────────────────────────────────
#
# Your Discord bot token. Used for bot-level API calls such as fetching which
# guilds the bot has joined.
#
# Where to get:
#   Discord Developer Portal (https://discord.com/developers/applications)
#   → Your Application → Bot → Reset Token
#
# ⚠ KEEP SECRET — treat this like a password. Never commit it or share it.
#
DISCORD_BOT_TOKEN=


# ── Discord OAuth2 ─────────────────────────────────────────────────────────────
#
# Client ID — the public identifier of your Discord application.
# Included in OAuth2 authorization URLs.
#
# Where to get:
#   Discord Developer Portal → Your Application → OAuth2 → Client ID
#
DISCORD_CLIENT_ID=

#
# Client Secret — used server-side to exchange authorization codes for tokens.
# Never expose this to the browser or include it in client-side code.
#
# Where to get:
#   Discord Developer Portal → Your Application → OAuth2 → Client Secret → Reset Secret
#
# ⚠ KEEP SECRET
#
DISCORD_CLIENT_SECRET=

#
# Redirect URI — the exact URL Discord will send users back to after they
# authorize your app. Must EXACTLY match one of the URLs registered in the
# Developer Portal (including protocol, domain, path, and no trailing slash).
#
# Where to configure:
#   1. Set this value to:  https://<your-api-domain>/api/auth/callback
#   2. Register the same URL at:
#      Discord Developer Portal → Your Application → OAuth2 → Redirects → Add Redirect
#
# Examples:
#   Development : https://abc123.sisko.replit.dev/api/auth/callback
#   Production  : https://api.lofiriya.com/api/auth/callback
#
DISCORD_REDIRECT_URI=


# ── MongoDB ────────────────────────────────────────────────────────────────────
#
# MongoDB connection string. Used to persist user data, guild settings,
# playlists, and sessions across restarts.
#
# Without this, the app runs with in-memory sessions (data is lost on restart
# but the app still works for testing).
#
# Where to get:
#   MongoDB Atlas (https://cloud.mongodb.com)
#   → Create a free M0 cluster → Connect → Drivers → Node.js
#   → Copy the connection string → replace <password> with your DB user password
#
# Example format:
#   mongodb+srv://username:password@cluster0.abcde.mongodb.net/lofiriya?retryWrites=true&w=majority
#
MONGODB_URI=


# ── Session / Auth ─────────────────────────────────────────────────────────────
#
# A long, random string used to cryptographically sign session cookies.
# Must stay secret. Changing this value will invalidate all existing sessions
# (all users will be logged out).
#
# Generate a strong value:
#   node -e "console.log(require('crypto').randomBytes(64).toString('hex'))"
#
# ⚠ KEEP SECRET — minimum 32 characters recommended, 64+ preferred.
#
SESSION_SECRET=


# ── Dashboard URLs ─────────────────────────────────────────────────────────────
#
# Public URL of your deployed frontend (dashboard website).
# Used for:
#   • Redirecting users to /dashboard after a successful OAuth2 login
#   • CORS allowlist (the API will only accept requests from this origin)
#
# Leave empty to have the API auto-detect the URL from the incoming request
# (works for development but is less reliable in production).
#
# Example: https://lofiriya-web--hostingnexus6.replit.app
#
FRONTEND_URL=

#
# Public URL of this API server.
# Used for health-check links and self-referencing documentation.
# Optional — not required for the app to function.
#
# Example: https://abc123.sisko.replit.dev
#
API_URL=


# ── Support Page ───────────────────────────────────────────────────────────────
#
# Discord webhook URL that receives bug reports and feature requests submitted
# via the support page. Without this, the support form will show an error when
# users try to submit.
#
# Where to get:
#   Discord Server → desired channel → Edit Channel (gear icon)
#   → Integrations → Webhooks → New Webhook → Copy Webhook URL
#
# ⚠ Keep reasonably secret — anyone with this URL can post to your channel.
#
SUPPORT_WEBHOOK_URL=


# ── Infrastructure ─────────────────────────────────────────────────────────────

#

# Port the API server listens on.
